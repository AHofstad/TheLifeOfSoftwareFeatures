+void one_click_print() {
+  ui.goto_screen([]{
+    char * const filename = card.longest_filename();
+    MenuItem_confirm::select_screen(
+      GET_TEXT_F(MSG_BUTTON_PRINT), GET_TEXT_F(MSG_BUTTON_CANCEL),

+      }, nullptr,
+      GET_TEXT_F(MSG_START_PRINT), filename, F("?")
+    );
+  });
