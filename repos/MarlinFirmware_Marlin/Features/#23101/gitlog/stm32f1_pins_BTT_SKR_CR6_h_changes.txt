+
+#define Y_ENABLE_PIN                        PB11
+#define Y_STEP_PIN                          PB10

+
+#define Z_ENABLE_PIN                        PB1

+#define Z_DIR_PIN                           PC5
+

+#define E0_STEP_PIN                         PB3
+#define E0_DIR_PIN                          PB4

+//

+//
+
