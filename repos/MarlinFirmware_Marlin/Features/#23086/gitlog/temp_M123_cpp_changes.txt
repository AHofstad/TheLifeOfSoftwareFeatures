+/**
+ * M123: Report fan states -or- set interval for auto-report
+ *
+ *   S<seconds> : Set auto-report interval

+
+  #if ENABLED(AUTO_REPORT_FANS)

+      fan_check.auto_reporter.set_interval(parser.value_byte());
+      return;
+    }
+  #endif
+
+  fan_check.print_fan_states();
+
+}
